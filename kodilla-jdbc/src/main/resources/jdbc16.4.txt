SELECT I.ID, I.SUMMARY AS ISSUES, IL.NAME AS ISSUES_LIST_NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUELIST_ID = IL.ID
ORDER BY I.ID;

SELECT I.ID, I.SUMMARY AS ISSUES, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY U.LASTNAME, U.FIRSTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
ORDER BY U.LASTNAME, U.FIRSTNAME;
